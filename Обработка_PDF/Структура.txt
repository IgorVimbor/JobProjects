Можно также добавить в код проверку наличия конфига и создание его при отсутствии:

def ensure_config_exists(self):
    """Проверка наличия и создание конфига при необходимости"""
    config_path = os.path.join(os.path.dirname(sys.executable), 'config.json')
    if getattr(sys, 'frozen', False):  # если это exe
        config_path = os.path.join(os.path.dirname(sys.executable), 'config.json')
    else:  # если запущено из Python
        config_path = 'config.json'

    if not os.path.exists(config_path):
        default_config = {
            "default_data": {
                "Номер акта": "",
                # ... остальные поля
            },
            "engine_models": [
                "534450",
                "53445.1000140-АЗ2",
                "5344.1000186"
            ],
            # ... остальные настройки
        }
        with open(config_path, 'w', encoding='utf-8') as f:
            json.dump(default_config, f, indent=4, ensure_ascii=False)
